SELECT M.nome, M.CPF
FROM Medicos M
JOIN Pacientes P ON M.CPF = P.CPF;

SELECT F.nome AS nome_funcionario, M.nome AS nome_medico
FROM Funcionarios F
JOIN Medicos M ON F.cidade = M.cidade;


SELECT M.nome, M.idade
FROM Medicos M
JOIN Consultas C ON M.codm = C.codm
JOIN Pacientes P ON C.codp = P.codp
WHERE P.nome = 'Ana';

SELECT A2.nroa
FROM Ambulatorios A1
JOIN Ambulatorios A2 ON A1.andar = A2.andar
WHERE A1.nroa = 5 AND A2.nroa <> 5;

SELECT DISTINCT codp, nome
FROM Pacientes
NATURAL JOIN Consultas
WHERE hora > '14:00';

SELECT DISTINCT nroa, andar
FROM Ambulatorios
NATURAL JOIN Medicos
NATURAL JOIN Consultas
WHERE data = '2020-10-12';

SELECT A.*, M.codm, M.nome
FROM Ambulatorios A
LEFT JOIN Medicos M ON A.nroa = M.nroa;

SELECT M.CPF, M.nome AS nome_medico, P.nome AS nome_paciente, C.data
FROM Medicos M
LEFT JOIN Consultas C ON M.codm = C.codm
LEFT JOIN Pacientes P ON C.codp = P.codp;
