Exercício 1:
CREATE VIEW FuncFloripa40 (codigo, nome, CPF, idade) AS
SELECT codf, nome, CPF, idade
FROM Funcionarios
WHERE cidade = 'Florianopolis' AND idade > 40
WITH CHECK OPTION;

Exercício 2

AS ON INSERT TO FuncFloripa40
DO INSTEAD
  INSERT INTO Funcionarios (codf, nome, CPF, idade, cidade)
